RAC

函数式编程思想：把一个操作写成一系列方法 链式编程思想：把要做的事情封装到block，给外界提供一个返回这个block的方法，方法返回值必须是block，block参数放需要操作的内容，block返回值是方法调用者

  • 使用场景：遍历数组或者字典
 NSArray *arr = @[@1,@2,@3];
 [arr.rac_sequence.signal subscribeNext:^(id x) {
        NSLog(@"%@",x);
    }];
NSDictionary *dict = @{@"account":@"aaa",@"name":@"xmg"};

[dict.rac_sequence.signal subscribeNext:^(RACTuple *x) {
        RACTupleUnpack(NSString *key, NSString *value) = x;
        NSLog(@"key:%@ value:%@",key, value);
    }];

  • 让请求按顺序执行

    RACSignal *signalA = [RACSignal createSignal:^RACDisposable *(id<RACSubscriber> subscriber) {
      NSLog(@"发送上部分请求");
      [subscriber sendNext:@"上部分数据"];
      [subscriber sendCompleted];
      return nil;
  }];
  RACSignal *signalB = [RACSignal createSignal:^RACDisposable *(id<RACSubscriber> subscriber) {
      NSLog(@"发送下部分请求");
      [subscriber sendNext:@"下部分数据"];
      return nil;
  }];
  //concat:按顺序去连接 第一个信号必须调用sendCompleted
  RACSignal *concatSignal = [signalA concat:signalB];
  [concatSignal subscribeNext:^(id x) {
      NSLog(@"%@",x);
  }];

  //压缩成一个信号
  //zipwith:当一个界面多个请求的时候，要等所有请求完成才能更新UI，要等所有信号都发送内容的时候才会调用
  RACSignal *zipSignal = [signalA zipWith:signalB];

  [zipSignal subscribeNext:^(id x) {
      NSLog(@"%@",x);
  }];
